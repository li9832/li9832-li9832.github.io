server配置

#
sysname server
#
ip pool pppoe-pool-1
 gateway-list 100.64.0.1 
 network 100.64.0.0 mask 255.255.255.0 
#
aaa
 local-user user99 password cipher pass_user99
 local-user user99 service-type ppp
#
interface Virtual-Template1
 ppp authentication-mode chap 
#   CHAP双向认证，只单向可不配置
# ppp chap user server11
# ppp chap password cipher  pass_server11
 remote address pool pppoe-pool-1
 ppp ipcp dns 223.5.5.5
 ip address 100.64.0.1 255.255.255.0 
#
interface GigabitEthernet0/0/0
 pppoe-server bind Virtual-Template 1
#

client配置

#
sysname client
#
acl number 2000  
 rule 5 permit 
#
interface Dialer1
 link-protocol ppp
# ppp authentication-mode chap    #单向认证不配置，双向需配置
 ppp chap user user99
 ppp chap password cipher pass_user99
 ip address ppp-negotiate
 ppp ipcp dns admit-any    #被动地接收PPPoE Server指定的DNS服务器地址。（可二选一）
 ppp ipcp dns request      #主动向PPPoE Server请求DNS服务器地址。执行命令display dns server，查看是否正确获取DNS服务器地址。
 dialer user ChinaTelecom  #随意配置，标记对端
 dialer bundle 1
 nat outbound 2000
#   CHAP双向认证，只单向可不配置
#aaa
# local-user server11 service-type ppp 
# local-user server11 password cipher pass_server11
#
interface GigabitEthernet0/0/0
 ip address 192.168.1.1 255.255.255.0 
#
interface GigabitEthernet0/0/1
 pppoe-client dial-bundle-number 1 
#
dialer-rule
 dialer-rule 1 ip permit
#
ip route-static 0.0.0.0 0.0.0.0 Dialer1
